const { createApp, ref, computed, onMounted } = Vue;


createApp({

    setup() {

        const currentTab = ref('schedule');

        const selectedDate = ref('2026-01-16');

        const tabs = [

            { id: 'schedule', icon: 'map-pin' },

            { id: 'info', icon: 'info' },

            { id: 'shop', icon: 'shopping-basket' },

            { id: 'expense', icon: 'banknote' }

        ];


        const dateList = [

            { full: '2026-01-16', weekday: 'Fri', day: '16' },

            { full: '2026-01-18', weekday: 'Sun', day: '18' },

            { full: '2026-01-19', weekday: 'Mon', day: '19' }

        ];


        const scheduleData = ref([

            { id: 1, date: '2026-01-16', category: '航班', name: 'CX450 TPE-NRT', time: '12:50' },

            { id: 2, date: '2026-01-18', category: '住宿', name: 'Hotel Aster', time: '15:00' }

        ]);


        const expenseList = ref([

            { id: 1, name: '新幹線 TOKI 325', amount: 6500, date: '2026-01-18' }

        ]);


        const filteredSchedule = computed(() => scheduleData.value.filter(s => s.date === selectedDate.value));

        const totalExpense = computed(() => expenseList.value.reduce((s, e) => s + e.amount, 0));


        const openNav = (name) => window.open(`https://www.google.com/maps/dir/?api=1&destination=2{encodeURIComponent(name)}`, '_blank');


        onMounted(() => { lucide.createIcons(); });


        return { currentTab, tabs, dateList, selectedDate, filteredSchedule, expenseList, totalExpense, openNav };

    }

}).mount('#app');

</script>
